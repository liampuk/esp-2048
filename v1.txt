#include <Wire.h>
#include <Adafruit_GFX.h>
#include <Adafruit_SSD1306.h>

#define SCREEN_WIDTH 128 // OLED width in pixels
#define SCREEN_HEIGHT 64 // OLED height in pixels

// Create OLED display object connected to I2C
Adafruit_SSD1306 oled(SCREEN_WIDTH, SCREEN_HEIGHT, &Wire, -1);

// Each character is 3x5 pixels, stored row by row (5 bytes per char)
// Here we define only digits 0–9 for simplicity
const uint8_t font3x5[][5] PROGMEM = {
    {0b111, 0b101, 0b101, 0b101, 0b111}, // '0'
    {0b010, 0b110, 0b010, 0b010, 0b111}, // '1'
    {0b111, 0b001, 0b111, 0b100, 0b111}, // '2'
    {0b111, 0b001, 0b111, 0b001, 0b111}, // '3'
    {0b101, 0b101, 0b111, 0b001, 0b001}, // '4'
    {0b111, 0b100, 0b111, 0b001, 0b111}, // '5'
    {0b111, 0b100, 0b111, 0b101, 0b111}, // '6'
    {0b111, 0b001, 0b010, 0b010, 0b010}, // '7'
    {0b111, 0b101, 0b111, 0b101, 0b111}, // '8'
    {0b111, 0b101, 0b111, 0b001, 0b111}  // '9'
};

// Draw a character at (x,y)
void drawChar3x5(int16_t x, int16_t y, char c, uint16_t color)
{
    if (c < '0' || c > '9')
        return; // only 0–9 defined
    const uint8_t *bitmap = font3x5[c - '0'];
    for (int row = 0; row < 5; row++)
    {
        uint8_t bits = pgm_read_byte(&bitmap[row]);
        for (int col = 0; col < 3; col++)
        {
            if (bits & (1 << (2 - col)))
            {
                oled.drawPixel(x + col, y + row, color);
            }
        }
    }
}

void clearBox(int x, int y)
{
    // Fill a 20x20 rectangle at (x,y) with black (background)
    oled.fillRect(x, y, 20, 20, SSD1306_BLACK);
}

void renderGrid()
{
    for (int y = 0; y < 4; y++)
    {
        for (int x = 0; x < 4; x++)
        {
            int top_x = 27 + x * 18;
            int top_y = 2 + y * 15;

            clearBox(top_x, top_y);
            oled.drawRect(top_x, top_y, 19, 16, SSD1306_WHITE); // outline rect
        }
    }
}

void setup()
{
    // Initialize OLED
    if (!oled.begin(SSD1306_SWITCHCAPVCC, 0x3C))
    { // 0x3C is default I2C addr
        Serial.println(F("SSD1306 allocation failed"));
        for (;;)
            ; // Don’t proceed, loop forever
    }

    oled.clearDisplay();
    oled.display();

    // Draw the grid once
    renderGrid();
    drawChar3x5(10, 10, '3', SSD1306_WHITE);
    drawChar3x5(14, 10, '7', SSD1306_WHITE);
    oled.display(); // Push buffer to screen
}

void loop()
{
    oled.clearDisplay();

    // Starting coordinates
    int x = 0;
    int y = 0;

    // Draw digits 0–9 one by one
    for (char c = '0'; c <= '9'; c++)
    {
        drawChar3x5(x, y, c, SSD1306_WHITE);
        oled.display(); // <--- update the screen right away
        x += 4;         // move 3px for glyph + 1px spacing
        // delay(100);     // now you’ll see them appear in sequence
    }

    delay(500); // pause before redrawing
}
