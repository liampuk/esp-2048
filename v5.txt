/**
 * Buttons
 */

struct Button
{
    int pin;
    bool state;    // debounced state
    bool lastRead; // last raw read
    unsigned long lastChange;
    unsigned long lastRepeat;
    bool pressedEventFired;
};

const unsigned long debounceDelay = 30; // ms
const unsigned long initialDelay = 500; // ms before auto-repeat
const unsigned long repeatDelay = 100;  // ms between repeats

Button wButton = {9, false, false, 0, 0, false};
Button aButton = {20, false, false, 0, 0, false};
Button sButton = {8, false, false, 0, 0, false};
Button dButton = {10, false, false, 0, 0, false};

int activeButtonPin = -1;
bool gameOver = false;

bool readButton(Button &btn)
{
    bool raw = digitalRead(btn.pin) == LOW;
    unsigned long now = millis();

    // debounce
    if (raw != btn.lastRead)
    {
        btn.lastChange = now;
        btn.lastRead = raw;
    }
    if ((now - btn.lastChange) > debounceDelay)
    {
        if (btn.state != raw)
        {
            btn.state = raw;
            if (btn.state)
            {
                // first press
                btn.lastRepeat = now;
                btn.pressedEventFired = false;
                activeButtonPin = btn.pin;
                return true; // fire immediately
            }
            else
            {
                // Released
                if (activeButtonPin == btn.pin)
                    activeButtonPin = -1;
            }
        }
    }

    // handle key repeat
    if (btn.state && activeButtonPin == btn.pin)
    {
        if (!btn.pressedEventFired)
        {
            if (now - btn.lastRepeat >= initialDelay)
            {
                btn.lastRepeat = now;
                btn.pressedEventFired = true;
                return true;
            }
        }
        else
        {
            if (now - btn.lastRepeat >= repeatDelay)
            {
                btn.lastRepeat = now;
                return true;
            }
        }
    }

    return false;
}

// Add random number to grid
void addRandom()
{
    int empty[16][2];
    int emptyCount = 0;
    for (int y = 0; y < 4; y++)
    {
        for (int x = 0; x < 4; x++)
        {
            if (grid[y][x] == 0)
            {
                empty[emptyCount][0] = x;
                empty[emptyCount][1] = y;
                emptyCount++;
            }
        }
    }
    if (emptyCount > 0)
    {
        int r = random(emptyCount);
        int value = (random(10) < 9) ? 2 : 4; // 90% chance of 2, 10% chance of 4
        grid[empty[r][1]][empty[r][0]] = value;
    }
}

void checkGameOver()
{
    // Check for any empty cells
    for (int y = 0; y < 4; y++)
    {
        for (int x = 0; x < 4; x++)
        {
            if (grid[y][x] == 0)
            {
                return; // Not game over, there's at least one empty cell
            }
        }
    }

    // Check for possible merges horizontally
    for (int y = 0; y < 4; y++)
    {
        for (int x = 0; x < 3; x++)
        {
            if (grid[y][x] == grid[y][x + 1])
            {
                return; // Not game over, there's a possible horizontal merge
            }
        }
    }

    // Check for possible merges vertically
    for (int x = 0; x < 4; x++)
    {
        for (int y = 0; y < 3; y++)
        {
            if (grid[y][x] == grid[y + 1][x])
            {
                return; // Not game over, there's a possible vertical merge
            }
        }
    }

    // If no empty cells and no possible merges, it's game over
    gameOver = true;
}

// Direction constants
#define DIR_UP 0
#define DIR_DOWN 1
#define DIR_LEFT 2
#define DIR_RIGHT 3

void shiftGrid(int dir)
{
    bool changed = false;
    for (int i = 0; i < 4; i++)
    {
        int lastMerge = (dir == DIR_UP || dir == DIR_LEFT) ? -1 : 4;
        for (int j = (dir == DIR_UP || dir == DIR_LEFT) ? 1 : 2;
             (dir == DIR_UP || dir == DIR_LEFT) ? j < 4 : j >= 0;
             (dir == DIR_UP || dir == DIR_LEFT) ? j++ : j--)
        {
            int *cell, *targetCell;
            int value, targetIdx = j;
            if (dir == DIR_UP || dir == DIR_DOWN)
            {
                cell = &grid[j][i];
                value = grid[j][i];
            }
            else
            {
                cell = &grid[i][j];
                value = grid[i][j];
            }
            if (value == 0)
                continue;
            // Find target
            while ((dir == DIR_UP && targetIdx > 0 && grid[targetIdx - 1][i] == 0) ||
                   (dir == DIR_DOWN && targetIdx < 3 && grid[targetIdx + 1][i] == 0) ||
                   (dir == DIR_LEFT && targetIdx > 0 && grid[i][targetIdx - 1] == 0) ||
                   (dir == DIR_RIGHT && targetIdx < 3 && grid[i][targetIdx + 1] == 0))
            {
                targetIdx += (dir == DIR_UP || dir == DIR_LEFT) ? -1 : 1;
            }
            // Merge
            if ((dir == DIR_UP && targetIdx > 0 && grid[targetIdx - 1][i] == value && lastMerge != targetIdx - 1) ||
                (dir == DIR_DOWN && targetIdx < 3 && grid[targetIdx + 1][i] == value && lastMerge != targetIdx + 1) ||
                (dir == DIR_LEFT && targetIdx > 0 && grid[i][targetIdx - 1] == value && lastMerge != targetIdx - 1) ||
                (dir == DIR_RIGHT && targetIdx < 3 && grid[i][targetIdx + 1] == value && lastMerge != targetIdx + 1))
            {
                if (dir == DIR_UP || dir == DIR_DOWN)
                {
                    grid[(dir == DIR_UP) ? targetIdx - 1 : targetIdx + 1][i] *= 2;
                    grid[j][i] = 0;
                    lastMerge = (dir == DIR_UP) ? targetIdx - 1 : targetIdx + 1;
                }
                else
                {
                    grid[i][(dir == DIR_LEFT) ? targetIdx - 1 : targetIdx + 1] *= 2;
                    grid[i][j] = 0;
                    lastMerge = (dir == DIR_LEFT) ? targetIdx - 1 : targetIdx + 1;
                }
                changed = true;
            }
            // Move
            else if (targetIdx != j)
            {
                if (dir == DIR_UP || dir == DIR_DOWN)
                {
                    grid[targetIdx][i] = value;
                    grid[j][i] = 0;
                }
                else
                {
                    grid[i][targetIdx] = value;
                    grid[i][j] = 0;
                }
                changed = true;
            }
        }
    }
    if (changed)
    {
        addRandom();
        checkGameOver();
    }
}

void drawChar3x5(int16_t x, int16_t y, char c, uint16_t color)
{
    if (c == '!')
        c = '['; // map ! to [ for char map
    if (c < '0' || c > '[')
    {
        return;
    }
    const uint8_t *bitmap = font3x5[c - '0'];
    for (int row = 0; row < 5; row++)
    {
        uint8_t bits = pgm_read_byte(&bitmap[row]);
        for (int col = 0; col < 3; col++)
        {
            if (bits & (1 << (2 - col)))
            {
                oled.drawPixel(x + col, y + row, color);
            }
        }
    }
}

void drawString3x5(int16_t x, int16_t y, const char *str, uint16_t color)
{
    while (*str)
    {
        drawChar3x5(x, y, *str++, color);
        x += 4; // 3 pixels wide + 1 pixel space
    }
}

void render_bg(int x, int y, int num)
{
    oled.fillRect(x + 1, y + 1, 17, 14, SSD1306_BLACK);
    // fill box with white dots
    for (int i = 0; i < 17; i += 1)
    {
        for (int j = 0; j < 14; j += 1)
        {
            bool shouldDraw = true;

            switch (num)
            {
            case 1024:
                shouldDraw = !(j % 2 == 0 && (i + j) % 4 == 0);
                break;
            case 512:
                shouldDraw = (i + j) % 5 != 0;
                break;
            case 256:
                shouldDraw = (i + j) % 4 != 0;
                break;
            case 128:
                shouldDraw = (i + j) % 3 != 0;
                break;
            case 64:
                shouldDraw = (i + j) % 2 == 0;
                break;
            case 32:
                shouldDraw = (i + j) % 3 == 0;
                break;
            case 16:
                shouldDraw = (i + j) % 4 == 0;
                break;
            case 8:
                shouldDraw = (i + j) % 5 == 0;
                break;
            case 4:
                shouldDraw = (j % 2 == 0 && (i + j) % 4 == 0);
                break;
            case 2:
            case 0:
                shouldDraw = false;
                break;
            }

            if (shouldDraw)
            {
                oled.drawPixel(x + 1 + i, y + 1 + j, SSD1306_WHITE);
            }

int char_offset_y = 6;

void renderGrid()

            int top_x = 27 + x * 18;
            int top_y = 2 + y * 15;

            render_bg(top_x, top_y, grid[y][x]);
            oled.drawRect(top_x, top_y, 19, 16, SSD1306_WHITE);
            if (grid[y][x] != 0)
            {
                if (grid[y][x] < 10)
                {
                    oled.fillRect(top_x + 7, top_y + char_offset_y - 1, 5, 7, SSD1306_BLACK);
                    drawChar3x5(top_x + 8, top_y + char_offset_y, '0' + grid[y][x], SSD1306_WHITE);
                }
                else if (grid[y][x] < 100)
                {
                    oled.fillRect(top_x + 5, top_y + char_offset_y - 1, 9, 7, SSD1306_BLACK);
                    drawChar3x5(top_x + 6, top_y + char_offset_y, '0' + grid[y][x] / 10, SSD1306_WHITE);
                    drawChar3x5(top_x + 10, top_y + char_offset_y, '0' + grid[y][x] % 10, SSD1306_WHITE);
                }
                else if (grid[y][x] < 1000)
                {

                    drawChar3x5(top_x + 4, top_y + char_offset_y, '0' + grid[y][x] / 100, SSD1306_WHITE);

{
    if (!oled.begin(SSD1306_SWITCHCAPVCC, 0x3C))

            ; // loop forever

    pinMode(wButton.pin, INPUT_PULLUP);
    pinMode(aButton.pin, INPUT_PULLUP);
    pinMode(sButton.pin, INPUT_PULLUP);

void loop()
{
    if (gameOver)
        showGameOver();
    else
    {
        if (readButton(wButton))
        {
            Serial.println("W");
            shiftGrid(DIR_UP);
            renderGrid();
            oled.display();
        }
        if (readButton(sButton))
        {
            Serial.println("S");
            shiftGrid(DIR_DOWN);
            renderGrid();
            oled.display();
        }
        if (readButton(aButton))
        {
            Serial.println("A");
            shiftGrid(DIR_LEFT);
            renderGrid();
            oled.display();
        }
        if (readButton(dButton))
        {
            Serial.println("D");
            shiftGrid(DIR_RIGHT);
            renderGrid();
            oled.display();
        }
    }
}
